            if (label.length <= 16) return label;

            const words = label.split(' ');

            const lines = [];

            let currentLine = words[0];

            for (let i = 1; i < words.length; i++) {

                if (currentLine.length + words[i].length + 1 <= 16) {

                    currentLine += ' ' + words[i];

                } else {

                    lines.push(currentLine);

                    currentLine = words[i];

                }

            }

            lines.push(currentLine);

            return lines;

        }

                    if (Array.isArray(label)) {

                        return label.join(' ');

                    } else {

                        return label;

                    }

                }

            }

        };
